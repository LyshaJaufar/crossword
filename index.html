import csv import json from pyodide.http import open_url import random taken = [] isTaken = True file = open_url("https://raw.githubusercontent.com/LyshaJaufar/crossword/master/data.json") data = json.load(file) for i in range(5): n = random.randint(1, len(data)) while isTaken: if len(taken) == 0: break for word in taken: if (word == data[n]["Words"]): n = random.randint(1, len(data)) if (word != data[n]["Words"]): isTaken = False break print(data[n]["Words"] + ": " + data[n]["Meaning"]) taken.append(data[n]["Words"]) isTaken = True i+=1
